## 1. HomePage.tsx

로그인한 사용자 정보를 표시하고 로그아웃 기능을 제공하는 메인 페이지.

```tsx
import { Container, Box, Typography, Button, Paper, Chip, Avatar } from "@mui/material"
import LogoutIcon from '@mui/icons-material/Logout';
import PersonIcon from '@mui/icons-material/Person';
import { useAuth } from "../store/authStore";
import { useNavigate } from "react-router-dom";

export default function HomePage() {
  const { user, logout } = useAuth(); // Context API로 전역 상태에서 user 정보 가져오기
  const navigate = useNavigate();

  const handleLogout = (): void => {
    logout();
    navigate('/login');
  }

  return (
    <Container maxWidth='sm'>
      <Box sx={{ mt: 8 }}>
        <Paper elevation={3} sx={{ p: 4, textAlign: 'center' }}>
          <Avatar sx={{ width: 80, height: 80, mx: 'auto', mb: 2, bgcolor: 'primary.main' }}>

          </Avatar>
          {/* user?.name : user가 null이면 오류 없이 undefined 반환 */}
          <Typography variant='h5' fontWeight='bold' gutterBottom>
            환영합니다. {user?.name}님!
          </Typography>

          <Box sx={{ textAlign: 'left', bgcolor: 'gray.50', p: 2, borderRadius: 1, mb: 3 }}>
            <Typography variant="body2" color='text.secondary'>이메일</Typography>
            <Typography variant="body1" fontWeight='medium'>{user?.email}</Typography>
            <Typography variant="body2" color='text.secondary'>권한</Typography>
            <Typography variant="body1" fontWeight='medium'>{user?.role}</Typography>
          </Box>

          <Button
            variant="outlined"
            color="error"
            startIcon={<LogoutIcon />}
            onClick={handleLogout}
            fullWidth
          >
            로그아웃
          </Button>
        </Paper>
      </Box>
    </Container>
  )
}
```

## 2. SignupPage.tsx

회원가입 폼. 유효성 검사 + useMutation + 구글 로그인 버튼 포함.

```tsx
import { useState } from "react"
import { useNavigate, Link } from "react-router-dom"
import { useMutation } from "@tanstack/react-query"
import { Container, Box, TextField, Button, Typography, Alert, Divider, CircularProgress, Paper } from "@mui/material"
import PersonAddIcon from '@mui/icons-material/PersonAdd';
import GoogleIcon from '@mui/icons-material/Google';
import { signupApi, googleLogin } from "../api/authApi";
import { useAuth } from "../store/authStore";
import type { SignupRequest, SignupFormErrors } from "../types/auth";
import type { ChangeEvent, FormEvent } from "react";

export default function SignupPage() {
  const navigate = useNavigate();
  const { login } = useAuth();

  const [form, setForm] = useState<SignupRequest>({
    email: '',
    password: '',
    name: ''
  });
  const [errors, setErrors] = useState<SignupFormErrors>({});

  // 유효성 검사 함수 — 오류가 없으면 true 반환
  const validate = (): boolean => {
    const newErrors: SignupFormErrors = {};

    if (!form.name.trim()) newErrors.name = '이름을 입력해주세요.';
    if (!form.email.trim()) newErrors.email = '이메일을 입력해주세요.';
    else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(form.email)) newErrors.email = '올바른 이메일 형식이 아닙니다.';
    if (!form.password) newErrors.password = '비밀번호를 입력해주세요.';
    else if (form.password.length < 8) newErrors.password = '비밀번호는 8자 이상이어야 합니다.';

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  }

  const signupMutation = useMutation({
    mutationFn: signupApi,
    onSuccess: data => {
      login(data.token, {
        email: data.email,
        name: data.name,
        role: data.role
      });
      navigate('/');
    },
  });

  const handleSubmit = (e: FormEvent<HTMLFormElement>): void => {
    e.preventDefault();
    if (validate()) {
      signupMutation.mutate(form);
    }
  };

  const handleChange = (e: ChangeEvent<HTMLInputElement>): void => {
    const { name, value } = e.target;
    setForm(prev => ({ ...prev, [name]: value }));
    // 입력 시 해당 필드 오류 메시지 초기화
    if (errors[name as keyof SignupFormErrors]) {
      setErrors(prev => ({ ...prev, [name]: '' }));
    }
  }

  return (
    <Container maxWidth="xs">
      <Box sx={{ mt: 8, display: 'flex', flexDirection: 'column', alignItems: 'center' }}>
        <Paper elevation={3} sx={{ p: 4, width: '100%' }}>
          <Box sx={{ display: 'flex', flexDirection: 'column', alignItems: 'center', mb: 3 }}>
            <PersonAddIcon color="primary" sx={{ fontSize: 48, mb: 1 }} />
            <Typography>회원가입</Typography>
          </Box>

          {/* 서버 오류 발생 시 Alert 표시 */}
          {signupMutation.isError && (
            <Alert severity="error" sx={{ mb: 2 }}>
              {/* error는 unknown 타입 → instanceof로 Error 인스턴스 확인 후 메시지 출력 */}
              {signupMutation.error instanceof Error
                ? signupMutation.error.message
                : '회원가입 중 오류가 발생했습니다.'}
            </Alert>
          )}

          <Box component='form' onSubmit={handleSubmit}>
            <TextField
              fullWidth label='이름' name='name' value={form.name}
              onChange={handleChange} error={!!errors.name}
              helperText={errors.name} margin='normal' autoFocus
            />
            <TextField
              fullWidth label='이메일' name='email' value={form.email}
              onChange={handleChange} error={!!errors.email}
              helperText={errors.email} margin='normal' type='email'
            />
            <TextField
              fullWidth label='비밀번호 (8자 이상)' name='password' value={form.password}
              onChange={handleChange} error={!!errors.password}
              helperText={errors.password} margin='normal' type='password'
            />
            {/* isPending 중에는 버튼 비활성화 + 로딩 스피너 표시 */}
            <Button
              type='submit' fullWidth variant="contained" size="large"
              sx={{ mt: 2 }} disabled={signupMutation.isPending}
            >
              {signupMutation.isPending ? <CircularProgress size={24} color="inherit" /> : '회원가입'}
            </Button>
          </Box>

          <Divider sx={{ my: 3 }}>
            <Typography variant="body2" color='text.secondary'>또는</Typography>
          </Divider>

          <Button
            fullWidth variant="contained" size="large"
            startIcon={<GoogleIcon />}
            onClick={googleLogin}
            sx={{ borderColor: '#4285f4', color: '#4285f4' }}
          >
            Google로 계속하기
          </Button>

          <Box sx={{ mt: 2, textAlign: 'center' }}>
            <Typography variant="body2" color='text.secondary'>
              이미 계정이 있으신가요?{' '}
              <Link to='/login' style={{ color: '#4285f4' }}>로그인</Link>
            </Typography>
          </Box>
        </Paper>
      </Box>
    </Container>
  )
}
```

### 새로 사용한 MUI 컴포넌트

| 컴포넌트 | 역할 |
|----------|------|
| `Alert` | 오류/성공 메시지 표시. `severity`로 색상 구분 (`error` / `success` / `warning`) |
| `Divider` | 구분선. 내부에 텍스트 삽입 가능 |
| `CircularProgress` | 로딩 스피너 |
| `TextField.helperText` | input 하단에 오류 메시지 표시 |
| `TextField.error` | input 테두리를 빨간색으로 변경 |

## 4. OAuth2CallbackPage.tsx

구글 로그인 완료 후 백엔드가 redirect해주는 페이지.  
쿼리스트링에서 JWT와 사용자 정보를 추출해서 저장한다.

```tsx
import { useEffect, useState } from "react"
import { useNavigate, useSearchParams } from "react-router-dom"
import { Box, CircularProgress, Typography, Alert } from "@mui/material"
import { useAuth } from "../store/authStore";

export default function OAuth2CallbackPage() {
  const navigate = useNavigate();
  const [searchParams] = useSearchParams();
  const { login } = useAuth();
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const token = searchParams.get('token');
    const email = searchParams.get('email');
    const name = searchParams.get('name');
    const role = searchParams.get('role') ?? 'ROLE_USER'; // null이면 기본값 ROLE_USER

    if (!token) {
      setError('로그인 중 오류가 발생했습니다. 다시 시도해주세요.');
      return;
    }

    // token이 존재하면 login 처리
    login(token, {
      email: email ?? '',   // null이면 빈 문자열
      name: name ?? '',
      role,
    });

    // replace: true — 뒤로가기 시 callback 페이지로 돌아오지 않도록
    navigate('/', { replace: true });
  }, [searchParams]);

  if (error) {
    return (
      <Box sx={{ mt: 8, textAlign: 'center' }}>
        <Alert severity="error" sx={{ maxWidth: 400, mx: 'auto' }}>{error}</Alert>
      </Box>
    )
  }

  return (
    <Box sx={{ mt: 8, display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 2 }}>
      <CircularProgress size={48} />
      <Typography color='text.secondary'>구글 로그인 처리중입니다.</Typography>
    </Box>
  )
}
```

### 쿼리스트링 (Query String)

백엔드가 구글 로그인 성공 후 redirect하는 URL 형태:
```
http://localhost:5173/oauth2/callback?token=xxx&email=xxx&name=xxx&role=ROLE_USER
```

`?` 뒤의 부분을 **쿼리스트링**이라고 한다.  
`key=value` 형태이며 `&`로 구분된다.

```ts
// useSearchParams()로 쿼리스트링 읽기
const [searchParams] = useSearchParams();
searchParams.get('token'); // ?token=xxx 에서 xxx 반환
searchParams.get('role');  // ?role=ROLE_USER 에서 ROLE_USER 반환
```

> Spring Boot에서 `@PathVariable`이 `/api/vehicles/1`의 `/1`을 가져오는 것처럼  
> `useSearchParams`는 `?` 뒤의 쿼리스트링 값을 가져온다.

### `??` (null 병합 연산자)

```ts
const role = searchParams.get('role') ?? 'ROLE_USER';
// searchParams.get('role')이 null/undefined면 'ROLE_USER' 사용

email: email ?? ''
// email이 null/undefined면 빈 문자열 사용
```

**`||` 와의 차이점**

| 연산자 | falsy 처리 대상 |
|--------|----------------|
| `\|\|` | `null`, `undefined`, `0`, `false`, `''` 모두 처리 |
| `??` | `null`, `undefined` **만** 처리 |

```ts
0 || 'default'   // → 'default' (0도 falsy로 처리)
0 ?? 'default'   // → 0 (null/undefined가 아니므로 그대로)
```

### `{ replace: true }`

```ts
navigate('/', { replace: true });
```
- 뒤로가기 버튼을 눌렀을 때 callback 페이지로 돌아오지 않도록 히스토리 스택에서 현재 페이지를 대체한다.
- 없으면 뒤로가기 시 callback 페이지로 돌아가 불필요한 요청이 발생할 수 있다.

## 5. App.tsx

```tsx
import { BrowserRouter, Routes, Route, Navigate } from "react-router-dom"
import { QueryClient, QueryClientProvider } from "@tanstack/react-query"
import { CssBaseline, ThemeProvider, createTheme } from "@mui/material"
import type { ReactNode } from "react"
import { AuthProvider, useAuth } from "./store/authStore"
import HomePage from "./pages/HomePages"
import LoginPage from "./pages/LoginPages"
import SignupPage from "./pages/SignupPage"
import OAuth2CallbackPage from "./pages/OAuth2CallbackPage"

const queryClient = new QueryClient({
  defaultOptions: {
    queries: { retry: 1, staleTime: 1000 * 60 } // 실패 시 1회 재시도, 1분간 fresh 유지
  },
});

const theme = createTheme({
  palette: { primary: { main: '#1976d2' } }
});

interface RouteGuardProps {
  children: ReactNode;
}

// 로그인한 사용자만 접근 가능 — 미로그인 시 /login으로 redirect
function PrivateRoute({ children }: RouteGuardProps) {
  const { isLoggedIn } = useAuth();
  return isLoggedIn ? <>{children}</> : <Navigate to='/login' replace />
}

// 비로그인 사용자만 접근 가능 — 로그인 상태면 /로 redirect
function GuestRoute({ children }: RouteGuardProps) {
  const { isLoggedIn } = useAuth();
  return isLoggedIn ? <Navigate to='/' replace /> : <>{children}</>
}

function AppRoutes() {
  return (
    <Routes>
      <Route path="/signup" element={<GuestRoute><SignupPage /></GuestRoute>} />
      <Route path="/login" element={<GuestRoute><LoginPage /></GuestRoute>} />
      <Route path="/oauth2/callback" element={<OAuth2CallbackPage />} />
      <Route path="/" element={<PrivateRoute><HomePage /></PrivateRoute>} />
      <Route path="*" element={<Navigate to='/' replace />} /> {/* 404 처리 */}
    </Routes>
  )
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <ThemeProvider theme={theme}>
        <CssBaseline />
        <BrowserRouter>
          <AuthProvider>
            <AppRoutes />
          </AuthProvider>
        </BrowserRouter>
      </ThemeProvider>
    </QueryClientProvider>
  )
}

export default App
```

### PrivateRoute / GuestRoute 패턴

| 컴포넌트 | 접근 가능 | 접근 불가 시 |
|----------|----------|------------|
| `PrivateRoute` | 로그인 사용자 | `/login`으로 redirect |
| `GuestRoute` | 비로그인 사용자 | `/`으로 redirect |

> 이전 수업에서 언급했던 **Protected Route** 패턴의 실제 구현이다.

### Provider 감싸기 순서

```tsx
<QueryClientProvider>      // React Query
  <ThemeProvider>           // MUI 테마
    <CssBaseline />         // 브라우저 CSS 초기화
    <BrowserRouter>         // React Router
      <AuthProvider>        // 인증 전역 상태
        <AppRoutes />
      </AuthProvider>
    </BrowserRouter>
  </ThemeProvider>
</QueryClientProvider>
```

> `useAuth()`는 `AuthProvider` 내부에서만 동작하므로  
> `AppRoutes`가 `AuthProvider` 안에 있어야 한다.

## 6. 자주 발생하는 오류 & 실수 모음

### 라우트 관련

**`useAuth()`를 `AuthProvider` 밖에서 호출**
```tsx
// ❌ AppRoutes가 AuthProvider 밖에 있으면 useAuth() 오류 발생
function App() {
  return (
    <BrowserRouter>
      <AppRoutes />       // useAuth() 사용 중인데 AuthProvider 밖
      <AuthProvider>...</AuthProvider>
    </BrowserRouter>
  )
}

// ✅ AppRoutes를 AuthProvider 안에 위치
<AuthProvider>
  <AppRoutes />
</AuthProvider>
```

**`replace: true` 누락**
```tsx
// ❌ 없으면 구글 로그인 후 뒤로가기 시 callback 페이지로 돌아감
navigate('/');

// ✅ 히스토리 스택 대체
navigate('/', { replace: true });
```

---

### 유효성 검사 관련

**`validate()` 호출 없이 바로 mutate**
```tsx
// ❌ 빈 값으로 서버에 요청이 가버림
const handleSubmit = (e) => {
  e.preventDefault();
  signupMutation.mutate(form);
}

// ✅ validate() 통과 후 mutate
const handleSubmit = (e) => {
  e.preventDefault();
  if (validate()) signupMutation.mutate(form);
}
```

**`error instanceof Error` 체크 누락**
```tsx
// ❌ error는 unknown 타입 → .message 접근 시 TS 오류
{signupMutation.error.message}

// ✅ instanceof로 타입 확인 후 접근
{signupMutation.error instanceof Error
  ? signupMutation.error.message
  : '오류가 발생했습니다.'}
```

---

### 쿼리스트링 관련

**`??` 대신 `||` 사용 시 의도치 않은 동작**
```ts
// ❌ || 는 0, false, '' 도 falsy 처리
const role = searchParams.get('role') || 'ROLE_USER';
// role이 빈 문자열('')이어도 'ROLE_USER'로 덮어씌워짐

// ✅ ?? 는 null/undefined만 처리
const role = searchParams.get('role') ?? 'ROLE_USER';
```

## 오늘 학습 요약

1. **HomePage** — `useAuth()`로 전역 상태에서 user 정보 조회, `user?.name` 옵셔널 체이닝
2. **SignupPage / LoginPage** — `useMutation` + 유효성 검사 + `CircularProgress` 로딩 처리
3. **OAuth2CallbackPage** — `useSearchParams()`로 쿼리스트링 파싱, `??` 연산자, `replace: true`
4. **App.tsx** — `PrivateRoute` / `GuestRoute` 패턴으로 페이지 접근 제어
5. **Provider 순서** — `QueryClient` → `ThemeProvider` → `BrowserRouter` → `AuthProvider` 순으로 감싸기
6. **쿼리스트링** — `?key=value&key=value` 형태, `useSearchParams().get()`으로 읽기
